#include <bits/stdc++.h>

struct Node {
    Node() = default;
    Node(int val) : val(val) {}
    int val {};
    Node* next {};
};

// 1 -> 2-> 3 -> 4
Node* init() {
    Node* cur = new Node(20);
    Node* cur_cur = cur;
    for (int i = 0; i < 10; i++) {
        Node* tmp = new Node(i);
        cur->next = tmp;
        cur = cur->next;
    }
    return cur_cur;
}

    if (!cur) {

        tmp = tmp->next;
    }

Node* reverse(Node* node) {
    if (node == nullptr) {
        return nullptr;
    }
    Node* pre { nullptr };
    Node* cur = node->next;
    while (cur) {
        Node* next = cur->next;
        cur->next = pre;
        pre = cur;
        cur = next;
    }
    node->next = pre;
    return node;
    // return pre;
}

void delete_list(Node* list) {
    if (!list) {
        return;
    }

    Node* cur = list;
    while (cur) {
        Node* next = cur->next;
        delete cur;
        cur = next;
    }
}
// 1->2->3->4->5
Node* dfs_reverse(Node* node) {
    if (node == nullptr) {
        return nullptr;
    }
    if (node && node->next == nullptr) {
        return node;
    }
    Node* cur = node;
    Node* f = dfs_reverse(node->next);
    Node* tmp = f;
    while (tmp->next) {
        tmp = tmp->next;
    }
    tmp->next = cur;
    cur->next = nullptr;
    return f;
}

Node* dfs(Node* list) {
    if (list == nullptr) {
        return nullptr;
    }

    Node* tmp = dfs_reverse(list->next);
    list->next = tmp;
    return list;
}
